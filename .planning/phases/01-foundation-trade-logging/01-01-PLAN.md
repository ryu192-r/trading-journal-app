  3. Write src/lib/validations/zodSchemas.ts:
     ```typescript
     import { z } from 'zod';
     import { Direction, SetupType } from '@/types/trade.types';
     
     export const DirectionSchema = z.enum(['LONG', 'SHORT']);
     export const SetupTypeSchema = z.enum(['EP', 'VCP', 'BREAKOUT', 'PULLBACK', 'PARABOLIC_LONG', 'CUSTOM']);
     
     export const TradeCreateSchema = z.object({
       symbol: z.string().min(1).max(20).toUpperCase(),
       direction: DirectionSchema,
       entryPrice: z.number().positive(),
       exitPrice: z.number().positive().optional().nullable(),
       stopPrice: z.number().positive().optional().nullable(),
       targetPrice: z.number().positive().optional().nullable(),
       quantity: z.number().int().positive().optional().nullable(),
       entryDate: z.date().optional().default(() => new Date()),
       exitDate: z.date().optional().nullable(),
       setupType: SetupTypeSchema.optional().nullable(),
       tags: z.array(z.string()).optional().default([]),
       notes: z.string().max(1000).optional().nullable(),
       screenshotUrl: z.string().url().optional().nullable(),
     });
     
     export const TradeUpdateSchema = TradeCreateSchema.partial();
     
     // For CSV row parsing (mapBrokerCSV output shape)
     export const CsvTradeSchema = TradeCreateSchema.omit({ entryDate: true }).extend({
       entryDate: z.string().or(z.date()).optional(),
     });
     
     // For natural language parsing
     export const ParseInputSchema = z.object({
       text: z.string().min(3).max(500),
     });
     ```
  4. The schema enforces D-30 (Zod) and D-31 (structured error format).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Auth middleware → Protected handlers | Token verification guards access |
| API handlers → Database | All writes/reads via Prisma with user-scoping |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-A | Spoofing | Auth token acceptance | mitigate | Require JWT_SECRET set; verify signature via jwt.verify |
| T-01-B | Tampering | Client-supplied trade data | mitigate | Zod schema validates and sanitizes all inputs before DB write |
| T-01-C | Information Disclosure | Error messages from validation | accept | Zod errors structured field-wise, do not expose internal stack traces |
| T-01-D | Denial of Service | Large/malicious payloads | accept | Next.js body size limits apply; payload validation via Zod rejects early |
| T-01-E | Elevation of Privilege | Unauthorized access | mitigate | requireAuth middleware on all protected routes; userId extracted from verified token |

</threat_model>
