     ```typescript
     import { parse } from 'csv-parse/sync';
     import { TradeInput, ParsedRow } from '@/types/trade.types';
     
     export function parseCSV(content: string): ParsedRow[] {
       const records = parse(content, {
         columns: true,       // first row is headers
         skip_empty_lines: true,
         trim: true,
         relax_column_count: true,
       }) as Record<string, string>[];
       
       return records.map((row, index) => ({
         success: true,
         trade: {
           symbol: String(row['Symbol'] || row['symbol'] || ''),
           direction: (row['Buy/Sell'] || row['transaction_type'] || row['Trade Type'] || '').toUpperCase() === 'SELL' || (row['Buy/Sell'] || '').toUpperCase() === 'SELL' ? 'SHORT' : 'LONG',
           entryPrice: parseFloat(row['Average Price'] || row['trade_price'] || row['Price'] || '0'),
           quantity: row['Quantity'] ? parseInt(row['Quantity'], 10) : undefined,
           entryDate: row['Trade Date'] || row['trade_date'] || row['Trades Date'] ? new Date(row['Trade Date'] || row['trade_date'] || (row['Trades Date'] + ' ' + row['Trades Time'])) : undefined,
         },
         rowNumber: index + 2, // account for header
       }));
     }
     ```

  ```typescript
  import { parseCSV } from './csvParser';
  import { TradeInput, ParsedRow } from '@/types/trade.types';
  
  type Broker = 'zerodha' | 'dhan' | 'angelone' | 'icicidirect';
  
  const BROKER_HEADERS: Record<Broker, string[][]> = {
    zerodha: [['Symbol','Trade Type','Buy/Sell','Quantity','Average Price','Trade Date','Product','Order Execution Time','Exchange']],
    dhan: [['symbol','transaction_type','quantity','trade_price','trade_date','order_id','exchange','segment']],
    angelone: [['Script','Buy/Sell','Quantity','Trades Date','Trades Time','Price','Net Amount']],
    icicidirect: [['Trade No','Symbol','Buy/Sell','Quantity','Trade Price','Trade Date','Order Number','Status']],
  };
  
  function detectBroker(headers: string[]): Broker | null {
    for (const [broker, patterns] of Object.entries(BROKER_HEADERS)) {
      for (const pattern of patterns) {
        if (pattern.every(h => headers.includes(h))) return broker as Broker;
      }
    }
    return null;
  }
  
  const BROKER_MAPPERS: Record<Broker, (row: Record<string, string>, lineNum: number) => ParsedRow> = {
    zerodha: (row) => ({
      success: true,
      trade: {
        symbol: row['Symbol'].replace(/\s+/g, '').toUpperCase(),
        direction: row['Buy/Sell'].toUpperCase() === 'SELL' ? 'SHORT' : 'LONG',
        entryPrice: parseFloat(row['Average Price']),
        quantity: parseInt(row['Quantity'], 10),
        entryDate: new Date(row['Trade Date']),
        setupType: (row['Product'] as any) || undefined,
        tags: [row['Product']],
        notes: 'Imported from Zerodha',
      },
    }),
    dhan: (row) => ({
      success: true,
      trade: {
        symbol: row['symbol'].toUpperCase(),
        direction: row['transaction_type'].toUpperCase().includes('BUY') ? 'LONG' : 'SHORT',
        entryPrice: parseFloat(row['trade_price']),
        quantity: parseInt(row['quantity'], 10),
        entryDate: new Date(row['trade_date']),
        tags: ['DHAN'],
      },
    }),
    angelone: (row) => ({
      success: true,
      trade: {
        symbol: row['Script'].toUpperCase(),
        direction: row['Buy/Sell'].toUpperCase() === 'SELL' ? 'SHORT' : 'LONG',
        entryPrice: parseFloat(row['Price']),
        quantity: parseInt(row['Quantity'], 10),
        entryDate: new Date(`${row['Trades Date']} ${row['Trades Time'] || '00:00:00'}`),
        notes: 'AngelOne trade',
      },
    }),
    icicidirect: (row) => ({
      success: true,
      trade: {
        symbol: row['Symbol'].toUpperCase(),
        direction: row['Buy/Sell'].toUpperCase() === 'SELL' ? 'SHORT' : 'LONG',
        entryPrice: parseFloat(row['Trade Price']),
        quantity: parseInt(row['Quantity'], 10),
        entryDate: new Date(row['Trade Date']),
      },
    }),
  };
  
  export function mapBrokerCSV(content: string): { trades: TradeInput[]; errors: string[] } {
    const raw = parseCSV(content);
    if (raw.length === 0) return { trades: [], errors: ['Empty CSV or invalid format'] };
    
    const headers = Object.keys(raw[0].trade || {});
    const broker = detectBroker(headers);
    if (!broker) {
      return { trades: [], errors: ['Unknown broker CSV format — supported: Zerodha, Dhan, AngelOne, ICICI Direct'] };
    }
    
    const mapper = BROKER_MAPPERS[broker];
    const trades: TradeInput[] = [];
    const errors: string[] = [];
    
    for (const row of raw) {
      const result = mapper(row.trade!, row.rowNumber);
      if (result.success && result.trade) {
        trades.push(result.trade);
      } else {
        errors.push(`Row ${row.rowNumber}: ${result.errors?.join(', ')}`);
      }
    }
    
    return { trades, errors };
  }
  ```

  1. Create src/app/api/import/route.ts as a POST handler (Next.js App Router):
     ```typescript
     import { NextRequest, NextResponse } from 'next/server';
     import { requireAuth } from '@/middleware/auth';
     import { prisma } from '@/lib/db';
     import { mapBrokerCSV } from '@/lib/parsers/brokerMappers';
     import { TradeCreateSchema } from '@/lib/validations/zodSchemas';
     
     export async function POST(request: NextRequest) {
       const user = await requireAuth();
       const formData = await request.formData();
       const file = formData.get('file') as File | null;
       
       if (!file) {
         return NextResponse.json({ error: 'No file uploaded' }, { status: 400 });
       }
       
       const content = await file.text();
       const { trades, errors } = mapBrokerCSV(content);
       
       if (trades.length === 0 && errors.length > 0) {
         return NextResponse.json({ error: 'Failed to parse CSV', details: errors }, { status: 422 });
       }
       
       let createdCount = 0;
       let failedCount = 0;
       const failedRows: string[] = [];
       
       await prisma.$transaction(async (tx) => {
         for (const tradeData of trades) {
           try {
             // Validate with Zod before insert
             const validated = TradeCreateSchema.parse({
               ...tradeData,
               userId: user.userId,
             });
             
             // Auto-compute P&L if exitPrice exists
             if (validated.exitPrice) {
               const qty = validated.quantity || 1;
               const diff = validated.direction === 'LONG'
                 ? validated.exitPrice - validated.entryPrice
                 : validated.entryPrice - validated.exitPrice;
               validated.pnl = diff * qty;
             }
             
             await tx.trade.create({ data: validated as any });
             createdCount++;
           } catch (err: any) {
             failedCount++;
             failedRows.push(`Row ${err.data?.userId ? 'unknown' : 'error'}: ${err.message}`);
           }
         }
       });
       
       return NextResponse.json({
         success: true,
         imported: createdCount,
         failed: failedCount,
         errors: failedCount > 0 ? failedRows : undefined,
         message: `Imported ${createdCount} trades${failedCount > 0 ? `, ${failedCount} failed` : ''}`,
       });
     }
     ```
  2. Ensure endpoint uses `requireAuth` middleware, Zod validation, and transactional safety.
  3. The endpoint returns a summary message (D-10) and lists row errors if any (D-10).
