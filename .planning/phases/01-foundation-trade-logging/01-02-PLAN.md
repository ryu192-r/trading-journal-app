  ```typescript
  import { TradeInput, ParsedRow } from '@/types/trade.types'
  import { parseCSV } from './csvParser'
  
  type Broker = 'zerodha' | 'dhan' | 'angelone' | 'icicidirect'
  
  interface ColumnMap {
    symbol: string[]
    direction: string[]
    entryPrice: string[]
    quantity: string[]
    entryDate: string[]
    exitPrice?: string[]
    stopPrice?: string[]
    targetPrice?: string[]
    commission?: string[]
    taxes?: string[]
  }
  
  const BROKER_HEADERS: Record<Broker, string[]> = {
    zerodha: ['Symbol', 'Trade Type', 'Buy/Sell', 'Quantity', 'Average Price', 'Trade Date', 'Order Execution Time', 'Exchange', 'Product', 'LTP', 'PnL'],
    dhan: ['symbol', 'transaction_type', 'quantity', 'trade_price', 'trade_date', 'order_id', 'exchange', 'segment'],
    angelone: ['Script', 'Buy/Sell', 'Quantity', 'Trades Date', 'Trades Time', 'Price', ' brokerage', 'stt', 'transaction', 'net amount'],
    icicidirect: ['Trade No', 'Symbol', 'Buy/Sell', 'Quantity', 'Trade Price', 'Trade Date', 'Order Number', 'Status'],
  }
  
  function detectBroker(headers: string[]): Broker | null {
    for (const [broker, brokerHeaders] of Object.entries(BROKER_HEADERS)) {
      const matches = brokerHeaders.filter(h => headers.includes(h)).length
      if (matches >= 4) return broker as Broker
    }
    return null
  }
  
  export function mapBrokerCSV(content: string): { trades: TradeInput[]; errors: string[] } {
    const raw = parseCSV(content)
    if (raw.length === 0) return { trades: [], errors: ['Empty CSV or invalid format'] }
    
    const headers = Object.keys(raw[0].trade || {})
    const broker = detectBroker(headers)
    
    if (!broker) {
      return { trades: [], errors: ['Unknown broker format — supported: Zerodha, Dhan, AngelOne, ICICI Direct'] }
    }
    
    const mapper = BROKER_MAPPERS[broker]
    const { trades, errors } = raw.reduce((acc, row, idx) => {
      const result = mapper(row.trade!, idx + 2)
      if (result.success) {
        acc.trades.push(result.trade!)
      } else {
        acc.errors.push(`Row ${idx + 2}: ${result.errors?.join(', ')}`)
      }
      return acc
    }, { trades: [] as TradeInput[], errors: [] as string[] })
    
    return { trades, errors }
  }
  
  const BROKER_MAPPERS: Record<Broker, (row: Record<string, string>, lineNum: number) => ParsedRow> = {
    zerodha: (row) => {
      const direction = row['Buy/Sell'].toUpperCase() === 'BUY' ? 'LONG' : 'SHORT'
      return {
        success: true,
        trade: {
          symbol: row['Symbol'].replace(/\s+/g, '').toUpperCase(),
          direction,
          entryPrice: parseFloat(row['Average Price']),
          quantity: parseInt(row['Quantity']),
          entryDate: new Date(row['Trade Date']),
          tags: [row['Product']], // CNC/MIS/NRML
          notes: `Trade via Zerodha`,
        },
        errors: [],
      }
    },
    dhan: (row) => {
      const direction = row['transaction_type'].toUpperCase().includes('BUY') ? 'LONG' : 'SHORT'
      return {
        success: true,
        trade: {
          symbol: row['symbol'].toUpperCase(),
          direction,
          entryPrice: parseFloat(row['trade_price']),
          quantity: parseInt(row['quantity']),
          entryDate: new Date(row['trade_date']),
          tags: ['DHAN'],
        },
      }
    },
    angelone: (row) => {
      const direction = row['Buy/Sell'].toUpperCase() === 'BUY' ? 'LONG' : 'SHORT'
      return {
        success: true,
        trade: {
          symbol: row['Script'].toUpperCase(),
          direction,
          entryPrice: parseFloat(row['Price']),
          quantity: parseInt(row['Quantity']),
          entryDate: new Date(row['Trades Date'] + ' ' + row['Trades Time']),
          notes: `AngelOne trade`,
        },
      }
    },
    icicidirect: (row) => {
      const direction = row['Buy/Sell'].toUpperCase() === 'BUY' ? 'LONG' : 'SHORT'
      return {
        success: true,
        trade: {
          symbol: row['Symbol'].toUpperCase(),
          direction,
          entryPrice: parseFloat(row['Trade Price']),
          quantity: parseInt(row['Quantity']),
          entryDate: new Date(row['Trade Date']),
        },
      }
    },
  }
  ```

     ```typescript
     import { TradeInput } from '@/types/trade.types'
     
     export interface ParseResult {
       success: boolean
       trade?: TradeInput
       error?: string
     }
     
     // Pattern: "bought HDFC at 730, stop 710, target 760"
     // Pattern: "sold INFY 3200, SL 3250, target 3100"
     // Pattern: "long RELIANCE 2400 qty 50"
     // Pattern: "short TCS 3500 100 shares"
     
     const LONG_PATTERN = /(?:bought|long|buy)\s+([A-Za-z0-9]+)\s+(?:at|@)?\s*(\d+(?:\.\d+)?)(?:\s*,?\s*stop\s+(\d+(?:\.\d+)?))?(?:\s*,?\s*target\s+(\d+(?:\.\d+)?))?(?:\s*,?\s*qty\s+(\d+))?/i
     const SHORT_PATTERN = /(?:sold|short|sell)\s+([A-Za-z0-9]+)\s+(?:at|@)?\s*(\d+(?:\.\d+)?)(?:\s*,?\s*SL\s+(\d+(?:\.\d+)?))?(?:\s*,?\s*target\s+(\d+(?:\.\d+)?))?(?:\s*,?\s*qty\s+(\d+))?/i
     
     export function parseNaturalLanguage(input: string): ParseResult {
       const trimmed = input.trim()
       
       // Try LONG pattern first
       let match = trimmed.match(LONG_PATTERN)
       if (match) {
         const [, symbol, price, stop, target, qty] = match
         return {
           success: true,
           trade: {
             symbol: symbol.toUpperCase(),
             direction: 'LONG',
             entryPrice: parseFloat(price),
             stopPrice: stop ? parseFloat(stop) : undefined,
             targetPrice: target ? parseFloat(target) : undefined,
             quantity: qty ? parseInt(qty) : undefined,
             notes: `NL entry: ${trimmed}`,
             tags: ['NL-Entry'],
           },
         }
       }
       
       // Try SHORT pattern
       match = trimmed.match(SHORT_PATTERN)
       if (match) {
         const [, symbol, price, stop, target, qty] = match
         return {
           success: true,
           trade: {
             symbol: symbol.toUpperCase(),
             direction: 'SHORT',
             entryPrice: parseFloat(price),
             stopPrice: stop ? parseFloat(stop) : undefined,
             targetPrice: target ? parseFloat(target) : undefined,
             quantity: qty ? parseInt(qty) : undefined,
             notes: `NL entry: ${trimmed}`,
             tags: ['NL-Entry'],
           },
         }
       }
       
       return {
         success: false,
         error: 'Could not parse. Try: "bought HDFC at 730, stop 710, target 760" or "sold INFY 3200, SL 3250"',
       }
     }
     
     // Additional helper: quick direction/time inference
     export function inferDirectionFromPhrase(phrase: string): 'LONG' | 'SHORT' | null {
       if (/bought|long|buy/i.test(phrase)) return 'LONG'
       if (/sold|short|sell/i.test(phrase)) return 'SHORT'
       return null
     }
     ```
  
  2. Pattern coverage for v1:
     - "bought HDFC at 730, stop 710, target 760" → LONG with stop/target

     - "long RELIANCE 2400" → LONG with just entry
     - "short INFY 1550 qty 100" → SHORT with quantity
     - "bought SBIN @ 450" → @ symbol instead of "at"
  
  3. Unmatched phrases return structured error with example format; Phase 2 enhancement will add LLM fallback for complex phrasings.

     ```typescript
     import { NextRequest, NextResponse } from 'next/server'
     import { requireAuth } from '@/middleware/auth'
     import { prisma } from '@/lib/db'
     import { mapBrokerCSV } from '@/lib/parsers/brokerMappers'
     
     export async function POST(request: NextRequest) {
       const user = await requireAuth()
       const formData = await request.formData()
       const file = formData.get('file') as File | null
       
       if (!file) {
         return NextResponse.json({ error: 'No file uploaded' }, { status: 400 })
       }
       
       const content = await file.text()
       const { trades, errors } = mapBrokerCSV(content)
       
       if (trades.length === 0 && errors.length > 0) {
         return NextResponse.json({ error: 'Failed to parse CSV', details: errors }, { status: 422 })
       }
       
       // Create trades in database within transaction
       let createdCount = 0
       let failedCount = 0
       const failedRows: string[] = []
       
       await prisma.$transaction(async (tx) => {
         for (const tradeData of trades) {
           try {
             // augment with userId and compute pnl if exitPrice present
             const enriched: any = {
               ...tradeData,
               userId: user.userId,
             }
             
             if (enriched.exitPrice) {
               const qty = enriched.quantity || 1
               const priceDiff = enriched.direction === 'LONG'
                 ? enriched.exitPrice - enriched.entryPrice
                 : enriched.entryPrice - enriched.exitPrice
               enriched.pnl = priceDiff * qty
             }
             
             await tx.trade.create({ data: enriched })
             createdCount++
           } catch (err: any) {
             failedCount++
             failedRows.push(`Row ${enriched.symbol || 'unknown'}: ${err.message}`)
           }
         }
       })
       
       return NextResponse.json({
         success: true,
         imported: createdCount,
         failed: failedCount,
         errors: failedRows,
         message: `Imported ${createdCount} trades${failedCount > 0 ? `, ${failedCount} failed` : ''}`,
       })
     }
     ```

     - Transaction ensures atomic batch: either all succeed or partial with error list
     - Errors captured per-row; entire import not rejected due to single bad row
     - Returns count summary and detailed errors for user action
  </action>

     ```typescript
     import { requireAuth } from '@/middleware/auth'
     import { prisma } from '@/lib/db'
     import { parseNaturalLanguage } from '@/lib/parsers/nlParser'
     
     export async function POST(request: NextRequest) {
       const user = await requireAuth()
       const { text } = await request.json()
       
       if (!text || typeof text !== 'string') {
         return NextResponse.json({ error: 'Text field required' }, { status: 400 })
       }
       
       const result = parseNaturalLanguage(text)
       
       if (!result.success || !result.trade) {
         return NextResponse.json(
           { error: result.error || 'Parse failed', example: 'Try: "bought HDFC at 730, stop 710, target 760"' },
           { status: 422 }
         )
       }
       
       // Auto-create trade in database
       const enriched: any = {
         ...result.trade,
         userId: user.userId,
       }
       
       if (enriched.exitPrice) {
         const qty = enriched.quantity || 1
         const priceDiff = enriched.direction === 'LONG'
           ? enriched.exitPrice - enriched.entryPrice
           : enriched.entryPrice - enriched.exitPrice
         enriched.pnl = priceDiff * qty
       }
       
       const trade = await prisma.trade.create({ data: enriched })
       
       return NextResponse.json({ trade }, { status: 201 })
     }
     ```
