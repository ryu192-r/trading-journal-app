Ghost trade definition:
- For each closed trade (exitPrice exists), look 30 calendar days forward
- Fetch NSE daily OHLC for that symbol via dataFetcher (Yahoo Finance ticker mapping already exists)
- Compute highest high (maxPrice30d) and lowest low (minPrice30d)
- LONG trade missedProfit = (maxPrice30d - exitPrice) * quantity
- SHORT trade missedProfit = (exitPrice - minPrice30d) * quantity
- missedR = missedProfit / (risk per share = |entry - stop| * quantity); if stop missing, skip
- Return { tradeId, symbol, direction, exitPrice, maxPrice30d, minPrice30d, missedProfit, missedR }

    export interface GhostResult {
      tradeId: string
      symbol: string
      direction: Direction
      exitPrice: number
      exitDate: Date
      maxPrice30d: number
      minPrice30d: number
      missedProfit: number
      missedR: number | null // null if stopPrice unknown
    }

    export async function computeGhostTrade(tradeId: string, userId: string): Promise<GhostResult | null> {
      // 1. Fetch closed trade with userId check
      const trade = await prisma.trade.findFirst({
        where: { id: tradeId, userId },
        select: { id: true, symbol: true, direction: true, entryPrice: true, exitPrice: true, stopPrice: true, quantity: true, exitDate: true }
      })
      if (!trade || !trade.exitPrice || !trade.exitDate) return null

      // 2. Determine ticker for Yahoo Finance
      const ticker = toYahooTicker(trade.symbol)

      // 3. Fetch 30 calendar days of OHLC starting day after exitDate
      const start = new Date(trade.exitDate)
      start.setDate(start.getDate() + 1)
      const end = new Date(trade.exitDate)
      end.setDate(end.getDate() + 31) // inclusive window covering 30 calendar days
      const candles = await getDailyOHLC(ticker, start, end)

      if (candles.length === 0) return null

      // 4. Compute maxPrice30d = max high over first 30 trading days (or calendar days? use available candles, cap at ~30)
      const window30 = candles.slice(0, 30)
      const highs = window30.map(c => c.high)
      const lows = window30.map(c => c.low)
      const maxPrice30d = Math.max(...highs)
      const minPrice30d = Math.min(...lows)

      // 5. Quantity default to 1
      const qty = trade.quantity || 1
      let missedProfit = 0
      if (trade.direction === 'LONG') {
        missedProfit = (maxPrice30d - trade.exitPrice) * qty
      } else {
        missedProfit = (trade.exitPrice - minPrice30d) * qty
      }

      // 6. Compute missedR if stop exists
      let missedR: number | null = null
      if (trade.stopPrice) {
        const riskPerShare = Math.abs(trade.entryPrice - trade.stopPrice)
        if (riskPerShare > 0) missedR = missedProfit / riskPerShare
      }

      return { ...trade, missedProfit, missedR }
    }
    ```

    import { NextRequest, NextResponse } from 'next/server'
    import { requireAuth, getUserIdFromToken } from '@/middleware/auth'
    import { computeGhostTrade } from '@/lib/analytics/calculator'

    export async function GET(request: NextRequest) {
      await requireAuth()
      const userId = await getUserIdFromToken()
      if (!userId) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

      const { searchParams } = new URL(request.url)
      const limit = parseInt(searchParams.get('limit') || '20')
      const offset = parseInt(searchParams.get('offset') || '0')

      // Get user's closed trades ordered by exitDate desc (most recent first)
      const closedTrades = await prisma.trade.findMany({
        where: { userId, exitPrice: { not: null } },
        select: { id: true },
        orderBy: { exitDate: 'desc' },
        skip: offset,
        take: limit,
      })

      const ghostPromises = closedTrades.trades.map(t => computeGhostTrade(t.id, userId))
      const ghosts = (await Promise.all(ghostPromises)).filter((g): g is NonNullable<typeof g> => g !== null)

      return NextResponse.json({
        ghosts,
        total: closedTrades.count,
        page: Math.floor(offset / limit) + 1,
        pageSize: limit,
      })
    }
    ```
